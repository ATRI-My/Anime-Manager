function calculateVisibleItems(totalItems, scrollTop, viewportHeight, itemHeight, overscan) {
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
  const endIndex = Math.min(
    totalItems,
    Math.max(startIndex, Math.ceil((scrollTop + viewportHeight) / itemHeight) + overscan)
  );
  return Math.max(0, endIndex - startIndex);
}
function shouldUseVirtualization(itemCount, threshold, enabled = true) {
  return enabled && itemCount > threshold;
}
